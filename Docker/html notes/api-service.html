server.js
import express from "express"; import mongoose from "mongoose"; const app = express(); app.use(express.json()); // Mongo connection mongoose.connect("mongodb://mongo-db:27017/demoDB", { useNewUrlParser: true, useUnifiedTopology: true, }); const UserSchema = new mongoose.Schema({ name: String, skill: String, }); const User = mongoose.model("User", UserSchema); app.get("/", (req, res) => { res.send("üöÄ API is running inside Docker!"); }); // Create user app.post("/users", async (req, res) => { const user = await User.create(req.body); res.json(user); }); // Get all users app.get("/users", async (req, res) => { const users = await User.find(); res.json(users); }); app.listen(3000, () => console.log("API running on port 3000"));

üü¢ Step-4: docker-compose.yml

THIS IS THE MAGIC üß†

Docker Compose orchestrates both containers and sets up networking automatically.

docker-compose.yml
services: api: build: . container_name: api-server ports: - "3000:3000" depends_on: - mongo-db networks: - devnet mongo-db: image: mongo container_name: mongo-db volumes: - mongoData:/data/db ports: - "27017:27017" networks: - devnet networks: devnet: volumes: mongoData:

API Service

  • build: . - Build from Dockerfile
  • ports: "3000:3000" - Expose API
  • depends_on: mongo-db - Start order
  • networks: devnet - Join network

MongoDB Service

  • image: mongo - Official image
  • volumes: mongoData - Persistent storage
  • ports: "27017:27017" - Expose DB
  • networks: devnet - Join network
