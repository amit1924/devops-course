üì¶ Deployment Configuration

App ‚Üí Container ‚Üí Pod ‚Üí Deployment

deployment.yaml 

apiVersion: apps/v1
kind: Deployment
metadata:
  name: express-app
spec:
  replicas: 3
  selector:
    matchLabels:
      app: express
  template:
    metadata:
      labels:
        app: express
    spec:
      containers:
      - name: express-container
        image: amit/express-app:v1
        ports:
        - containerPort: 3000

Apply: kubectl apply -f deployment.yaml
